\subsection*{Section A: Multi-Concept (5 Questions)}
\begin{enumerate}[label=Q\arabic*.]

(B) Enhanced immunity + anemia

(D) Excessive sweating + hyperthyroidism

\subsection*{Section D: Matching (2 Questions)}
\begin{enumerate}[label=Q\arabic*.]
\setcounter{enumi}{13}
\item Match the following (NCERT based):  
1. PCR $\rightarrow$ (a) DNA amplification  
2. ELISA $\rightarrow$ (b) Antigen-antibody detection  
3. Western blot $\rightarrow$ (c) Protein identification  
4. Northern blot $\rightarrow$ (d) RNA detection  

Correct match:  
(A) 1-a, 2-b, 3-c, 4-d  
(B) 1-b, 2-a, 3-d, 4-c  
(C) 1-c, 2-d, 3-a, 4-b  
(D) 1-a, 2-c, 3-b, 4-d

\item Match the following hormones with functions:  
1. Insulin $\rightarrow$ (a) Lowers blood glucose  
2. Glucagon $\rightarrow$ (b) Raises blood glucose  
3. Thyroxine $\rightarrow$ (c) Increases basal metabolic rate  
4. ADH $\rightarrow$ (d) Promotes water reabsorption  

Correct match:  
(A) 1-a, 2-b, 3-c, 4-d  
(B) 1-b, 2-a, 3-d, 4-c  
(C) 1-c, 2-d, 3-a, 4-b  
(D) 1-a, 2-c, 3-b, 4-d
\end{enumerate}
